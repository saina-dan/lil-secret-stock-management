⚙️ Setup Guide
1
Install PostgreSQL
Download from postgresql.org or use your system package manager.
sudo apt install postgresql postgresql-contrib
2
Create the database
Run in psql or pgAdmin:
CREATE DATABASE lil_secret;
3
Install & configure backend
Download the backend zip, set your .env, then run:
npm install cp .env.example .env # Edit .env with your DB password + JWT secrets npm run migrate # apply schema npm run seed # create admin/staff users npm start # start on :3001
4
Login to activate backend sync
Log in with your admin password — the system auto-detects the backend and syncs all data from PostgreSQL in real-time.
🐘 Why PostgreSQL
ACID Transactions Stock updates & payments are atomic — no partial writes
Row-level Locking FOR UPDATE prevents double-booking & race conditions
Generated Columns closing & hours_worked always computed by the DB
Audit Triggers Every change auto-logged with timestamp + IP
Fuzzy Search pg_trgm indexes for instant item & guest search
Connection Pool 20-connection pool handles concurrent staff users

📘 Backend Integration Guide
🟠 AWS S3 — Node.js (multer-s3)
npm install @aws-sdk/client-s3 multer-s3

// server.js
const { S3Client } = require('@aws-sdk/client-s3');
const multerS3  = require('multer-s3');
const multer    = require('multer');

const s3 = new S3Client({ region: process.env.AWS_REGION });
const upload = multer({
  storage: multerS3({
    s3, bucket: process.env.S3_BUCKET,
    key: (req, file, cb) =>
      cb(null, `${req.body.type}/${Date.now()}-${file.originalname}`)
  })
});

app.post('/storage/upload', upload.single('file'), (req, res) => {
  res.json({ url: req.file.location, key: req.file.key });
});
⚡ Supabase Storage — JS SDK
npm install @supabase/supabase-js

import { createClient } from '@supabase/supabase-js';
const supabase = createClient(SUPABASE_URL, SUPABASE_ANON_KEY);

async function uploadFile(file, category) {
  const path = `${category}/${Date.now()}-${file.name}`;
  const { data, error } = await supabase.storage
    .from('lil-secret')
    .upload(path, file, { upsert: false });
  if (error) throw error;
  const { data: { publicUrl } } =
    supabase.storage.from('lil-secret').getPublicUrl(path);
  return { path, publicUrl };
}
